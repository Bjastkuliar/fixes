# Do a SQL Query
POST /_xpack/sql?format=txt
{
  "query": "select * from user_mapping",
  "fetch_size": 10
}

# Translate SQL to ES Query Language
GET /_xpack/sql/translate
{
  "query": "select Name, Result, Type from veeam_job"
}
